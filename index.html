require('dotenv').config(); const express = require('express'); const sqlite3 = require('sqlite3').verbose(); const cors = require('cors'); const axios = require('axios'); const Stripe = require('stripe'); const app = express(); const stripe = Stripe(process.env.STRIPE_SECRET_KEY || 'sk_test_mock_key'); app.use(cors()); app.use(express.json()); // ---------------------------------------------------- // 1. Initialisation de la Base de Données SQLite // ---------------------------------------------------- const db = new sqlite3.Database('./database.db', (err) => { if (err) console.error('Erreur BDD:', err.message); else console.log('Base de données SQLite connectée.'); }); db.serialize(() => { db.run(`CREATE TABLE IF NOT EXISTS users ( id INTEGER PRIMARY KEY AUTOINCREMENT, email TEXT UNIQUE, balance REAL DEFAULT 142.80 )`); db.run(`INSERT OR IGNORE INTO users (id, email, balance) VALUES (1, 'demo@boostvite.com', 142.80)`); db.run(`CREATE TABLE IF NOT EXISTS orders ( id INTEGER PRIMARY KEY AUTOINCREMENT, user_id INTEGER, service_id INTEGER, service_name TEXT, link TEXT, quantity INTEGER, price REAL, status TEXT DEFAULT 'En attente', api_order_id INTEGER, created_at DATETIME DEFAULT CURRENT_TIMESTAMP )`); }); // ---------------------------------------------------- // 2. Endpoints API (Backend) // ---------------------------------------------------- // Obtenir le profil utilisateur app.get('/api/user/:id', (req, res) => { db.get('SELECT id, email, balance FROM users WHERE id = ?', [req.params.id], (err, user) => { if (err || !user) return res.status(404).json({ error: 'Utilisateur introuvable' }); res.json(user); }); }); // Récupérer les commandes app.get('/api/orders/user/:userId', (req, res) => { db.all('SELECT * FROM orders WHERE user_id = ? ORDER BY id DESC', [req.params.userId], (err, rows) => { if (err) return res.status(500).json({ error: err.message }); res.json(rows); }); }); // Créer une commande app.post('/api/orders/create', async (req, res) => { const { userId, serviceId, serviceName, link, quantity, price } = req.body; db.get('SELECT balance FROM users WHERE id = ?', [userId], async (err, user) => { if (err || !user) return res.status(400).json({ error: 'Utilisateur introuvable' }); if (user.balance < price) return res.status(400).json({ error: 'Solde insuffisant' }); try { let providerOrderId = null; if (process.env.SMM_PROVIDER_API_KEY) { const response = await axios.post(process.env.SMM_PROVIDER_URL, new URLSearchParams({ key: process.env.SMM_PROVIDER_API_KEY, action: 'add', service: serviceId, link: link, quantity: quantity })); if (response.data && response.data.order) { providerOrderId = response.data.order; } } const newBalance = user.balance - price; db.run('UPDATE users SET balance = ? WHERE id = ?', [newBalance, userId]); db.run( `INSERT INTO orders (user_id, service_id, service_name, link, quantity, price, api_order_id) VALUES (?, ?, ?, ?, ?, ?, ?)`, [userId, serviceId, serviceName, link, quantity, price, providerOrderId], function(err) { if (err) return res.status(500).json({ error: err.message }); res.json({ success: true, orderId: this.lastID, newBalance: newBalance, providerOrderId: providerOrderId }); } ); } catch (apiError) { console.error('Erreur API Fournisseur:', apiError.message); res.status(500).json({ error: 'Erreur de transmission au fournisseur SMM' }); } }); }); // Session Stripe app.post('/api/payments/create-checkout-session', async (req, res) => { const { amount, userId } = req.body; try { const session = await stripe.checkout.sessions.create({ payment_method_types: ['card'], line_items: [{ price_data: { currency: 'usd', product_data: { name: 'Recharge Solde BoostVite' }, unit_amount: Math.round(amount * 100), }, quantity: 1, }], mode: 'payment', success_url: `http://localhost:3000/?payment=success`, cancel_url: `http://localhost:3000/?payment=cancel`, metadata: { userId, amount } }); res.json({ id: session.id, url: session.url }); } catch (e) { res.status(500).json({ error: e.message }); } }); // ---------------------------------------------------- // 3. `); }); // ---------------------------------------------------- // Démarrage // ---------------------------------------------------- const PORT = process.env.PORT || 3000; app.listen(PORT, () => { console.log(`Serveur démarré : http://localhost:${PORT}`); });
